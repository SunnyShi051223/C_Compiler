// ============================================
// 冒泡排序完整测试
// 验证：数组、嵌套循环、条件判断、set/get
// ============================================

int arr[10] ;
int n ;
int i ;
int j ;
int temp ;

// 初始化数组（逆序排列，方便验证排序正确性）
n = 10 ;
set(arr, 0, 9) ;
set(arr, 1, 8) ;
set(arr, 2, 7) ;
set(arr, 3, 6) ;
set(arr, 4, 5) ;
set(arr, 5, 4) ;
set(arr, 6, 3) ;
set(arr, 7, 2) ;
set(arr, 8, 1) ;
set(arr, 9, 0) ;

// ===== 输出排序前的数组 =====
char str[6] ;
set(str, 0, 'B') ;
set(str, 1, 'e') ;
set(str, 2, 'g') ;
set(str, 3, 'i') ;
set(str, 4, 'n') ;
set(str, 5, 0) ;

writeChar(get(str, 0)) ;
writeChar(get(str, 1)) ;
writeChar(get(str, 2)) ;
writeChar(get(str, 3)) ;
writeChar(get(str, 4)) ;

println() ;

i = 0 ;
while ( i < n ) {
    write(get(arr, i)) ;
    i = i + 1 ;
} ;

// ===== 冒泡排序 =====
// 外层循环：控制排序轮数
i = 0 ;
while ( i < 9 ) {
    // 内层循环：比较相邻元素
    j = 0 ;
    while ( j < 9 ) {
        // 如果前一个元素大于后一个，交换
        if ( get(arr, j) > get(arr, j + 1) ) {
            temp = get(arr, j) ;
            set(arr, j, get(arr, j + 1)) ;
            set(arr, j + 1, temp) ;
        } ;
        j = j + 1 ;
    } ;
    i = i + 1 ;
} ;

println() ;

// ===== 输出排序后的数组 =====
char endStr[4] ;
set(endStr, 0, 'E') ;
set(endStr, 1, 'n') ;
set(endStr, 2, 'd') ;
set(endStr, 3, 0) ;

writeChar(get(endStr, 0)) ;
writeChar(get(endStr, 1)) ;
writeChar(get(endStr, 2)) ;



println() ;

i = 0 ;
while ( i < n ) {
    write(get(arr, i)) ;
    i = i + 1 ;
} ;

return 0 ;
