## 1. 项目概述

本项目是一个基于简单优先法的微型程序语言（mico C）编译器，能够将源程序编译成Hack虚拟机的VM指令。编译器采用模块化设计，主要包括以下组件：

- **词法分析器（Lexer）**：将源程序字符序列转换为Token序列
- **文法处理器（Grammar）**：定义文法，计算FIRSTVT/LASTVT集合，构建优先关系表
- **语法分析器（PrecedenceParser）**：使用简单优先法进行语法分析
- **语义分析器（SemanticAnalyzer）**：进行语义检查，生成四元式中间代码
- **代码生成器（CodegenVisitor）**：遍历AST，生成VM指令
- **VM翻译器（VMTranslator）**：将VM指令翻译成汇编代码

```cpp
Token Lexer::nextToken() {
    // 1. 跳过空白字符和注释
    skipWhitespaceAndComments();

    // 2. 检查是否到达文件末尾
    if (isAtEnd()) return {END_OF_FILE, "", line_, column_};

    // 3. 根据当前字符类型调用相应的处理函数
    char c = currentChar();
    if (isDigit(c)) return readNumber();           // 数字
    if (c == '\'') return readCharLiteral();        // 字符常量
    if (isAlpha(c)) return readIdentifierOrKeyword(); // 标识符/关键字
    if (isOperator(c)) return readOperator();       // 运算符
    if (isDelimiter(c)) return readDelimiter();     // 分隔符

    // 4. 未知字符，报告错误
    errorHandler_.reportError(...);
    return {UNKNOWN, string(1, c), line_, column_};
}
```

### 5.5 FIRSTVT集合计算算法

```cpp
void Grammar::computeFirstVT() {
    // 初始化
    for (const auto& nt : nonTerminals_)
        firstVT_[nt] = set<string>();

    bool changed = true;
    while (changed) {
        changed = false;
        for (const auto& prod : productions_) {
            const string& A = prod.left;
            const auto& rhs = prod.right;

            // 规则1: 若有 A → a... 或 A → Ba...
            if (isTerminal(rhs[0]))
                if (firstVT_[A].insert(rhs[0]).second) changed = true;
            if (rhs.size() >= 2 && isNonTerminal(rhs[0]) && isTerminal(rhs[1]))
                if (firstVT_[A].insert(rhs[1]).second) changed = true;

            // 规则2: 若有 A → B...，且 a ∈ FIRSTVT(B)
            if (isNonTerminal(rhs[0]))
                for (const auto& a : firstVT_[rhs[0]])
                    if (firstVT_[A].insert(a).second) changed = true;
        }
    }
}
```

### 5.6 优先关系表构建算法

```cpp
void Grammar::buildPrecedenceTable() {
    // 遍历所有产生式
    for (const auto& prod : productions_) {
        const auto& rhs = prod.right;
        for (size_t i = 0; i + 1 < rhs.size(); i++) {
            const string& Xi = rhs[i];
            const string& Xi1 = rhs[i + 1];

            // 规则1: 若有 A → ...ab...，则 a = b
            if (isTerminal(Xi) && isTerminal(Xi1))
                relation_[Xi][Xi1] = "=";

            // 规则2: 若有 A → ...aB...，c ∈ FIRSTVT(B)，则 a <· c
            if (isTerminal(Xi) && isNonTerminal(Xi1))
                for (const auto& a : firstVT_[Xi1])
                    relation_[Xi][a] = "<";

            // 规则3: 若有 A → ...Ba...，c ∈ LASTVT(B)，则 c ·> a
            if (isNonTerminal(Xi) && isTerminal(Xi1))
                for (const auto& a : lastVTMap_[Xi])
                    relation_[a][Xi1] = ">";
        }
    }

    // 特殊关系
    relation_["$"][firstVT_["P"]] = "<";  // $ < FIRSTVT(P)
    relation_[lastVTMap_["P"]]["$"] = ">"; // LASTVT(P) > $
}
```

### 6.4 核心算法：简单优先法

```cpp
bool PrecedenceParser::parse(const vector<Token>& tokens) {
    // 初始化
    push({"$", true, 0, ...});  // 将$压入栈
    size_t inputPos = 0;

    while (inputPos < tokens.size()) {
        Token currentToken = tokens[inputPos];
        string currentTerminal = tokenToTerminal(currentToken);

        // 找到栈顶终结符
        int topTermPos = findTopTerminal();
        string topTerminal = stack_[topTermPos].symbol;

        // 获取优先关系
        string relation = grammar_.getRelation(topTerminal, currentTerminal);

        if (relation == "<" || relation == "=") {
            // 移进：将当前输入符号压入栈
            push({currentTerminal, true, ...});
            inputPos++;
        } else if (relation == ">") {
            // 归约：找到可归约的子串并归约
            if (!reduce()) {
                // 归约失败，报告错误
                errorHandler_.reportError(...);
                inputPos++;
            }
        } else {
            // 无关系，报告错误
            errorHandler_.reportError(...);
            inputPos++;
        }
    }

    // 输入结束，归约剩余符号
    while (true) {
        if (stack_.size() == 2 && stack_[0].symbol == "$" &&
            stack_[1].symbol == grammar_.getStartSymbol()) {
            return true;  // 分析成功
        }
        if (!reduce()) return false;  // 归约失败
    }
}
```

### 6.5 归约算法

```cpp
bool PrecedenceParser::reduce() {
    // 从栈顶开始，尝试找到可归约的子串
    for (int len = stack_.size(); len >= 1; len--) {
        int start = stack_.size() - len;
        vector<string> candidate;
        for (int i = start; i < stack_.size(); i++)
            candidate.push_back(stack_[i].symbol);

        // 遍历所有产生式，查找匹配的产生式
        for (const auto& prod : grammar_.getProductions()) {
            if (prod.right == candidate) {
                // 执行语义动作
                semanticAction(prod, start);
                // 构建AST节点
                buildASTForProduction(prod, start);
                // 弹出归约的符号
                for (int i = 0; i < len; i++) stack_.pop_back();
                // 压入归约后的非终结符
                push({prod.left, false, ...});
                return true;
            }
        }
    }
    return false;  // 没有找到可归约的产生式
}
```

### 6.6 语义动作示例

```cpp
void PrecedenceParser::semanticAction(const Production& prod, int start) {
    const auto& rhs = prod.right;

    // 赋值语句: S → id = E
    if (len == 3 && rhs[0] == "id" && rhs[1] == "=") {
        sem_.checkDeclared(stack_[start].addr, stack_[start].line);
        sem_.emit("=", stack_[start + 2].addr, "", stack_[start].addr);
    }

    // 二元运算: E → E op E
    if (len == 3 && rhs[0] == "E" && rhs[2] == "E") {
        string t = sem_.newTemp();
        sem_.emit(stack_[start + 1].symbol, stack_[start].addr, stack_[start + 2].addr, t);
        stack_[start].addr = t;
    }

    // while循环: S → while(E){P}
    if (len == 7 && rhs[0] == "while") {
        string L0 = sem_.newLabel(), L1 = sem_.newLabel();
        sem_.emit("label", L0, "", "");
        sem_.emit("if_false", cond, "", L1);
        // ... 循环体 ...
        sem_.emit("goto", L0, "", "");
        sem_.backpatch(j, L1);
        sem_.emit("label", L1, "", "");
    }
}
```

### 7.4 作用域管理

```cpp
// 作用域栈
vector<Scope> scopeStack_;

void SemanticAnalyzer::enterScope() {
    scopeStack_.push_back({{}, varCount_});
}

void SemanticAnalyzer::exitScope() {
    if (scopeStack_.size() > 1) {
        scopeStack_.pop_back();
    }
}

SymbolEntry* SemanticAnalyzer::lookup(const string& name) {
    // 从内到外查找
    for (int i = scopeStack_.size() - 1; i >= 0; i--) {
        auto it = scopeStack_[i].symbols.find(name);
        if (it != scopeStack_[i].symbols.end()) return &it->second;
    }
    return nullptr;
}
```

### 7.6 类型检查

```cpp
void SemanticAnalyzer::checkAssignment(const string& lhs, const string& rhs, int line) {
    string lhsType = getType(lhs);
    string rhsType = getType(rhs);

    // 类型相同，OK
    if (lhsType == rhsType) return;

    // int ← char : 警告
    if (lhsType == "int" && rhsType == "char") {
        cerr << "Type Warning: assigning char to int\n";
    }
    // double ← int : 允许（隐式转换）
    else if (lhsType == "double" && rhsType == "int") {
        cerr << "Type Warning: implicit conversion\n";
    }
    // 类型不兼容
    else {
        cerr << "Type Error: incompatible types\n";
    }
}
```

### 7.7 VM代码生成

```cpp
void SemanticAnalyzer::dumpTarget(const string& filename) const {
    ofstream f(filename);
    f << "function Main.main " << numLocals << "\n";

    for (const auto& q : quads_) {
        // 赋值: (=, src, _, dst)
        if (q.op == "=") {
            f << "push " << toVM(q.arg1) << "\n";
            f << "pop " << toVM(q.result) << "\n";
        }
        // 算术运算: (op, a, b, result)
        else if (q.op == "+" || q.op == "-" || ...) {
            f << "push " << toVM(q.arg1) << "\n";
            f << "push " << toVM(q.arg2) << "\n";
            f << "add\n";  // 或 sub, eq, gt, lt...
            f << "pop " << toVM(q.result) << "\n";
        }
        // 函数调用: (call, name, nArgs, result)
        else if (q.op == "call") {
            f << "call " << q.arg1 << " " << q.arg2 << "\n";
            f << "pop " << toVM(q.result) << "\n";
        }
        // 标签: (label, name, _, _)
        else if (q.op == "label") {
            f << "label " << q.arg1 << "\n";
        }
        // 跳转: (goto, label, _, _)
        else if (q.op == "goto") {
            f << "goto " << q.arg1 << "\n";
        }
        // 条件跳转: (if_false, cond, _, label)
        else if (q.op == "if_false") {
            f << "push " << toVM(q.arg1) << "\n";
            f << "not\n";
            f << "if-goto " << q.result << "\n";
        }
    }
}
```

### 8.3 访问者模式

```cpp
void CodegenVisitor::visit(const shared_ptr<ASTNode>& node) {
    if (!node) return;

    switch (node->type) {
        case ASTNodeType::Program:       visitProgram(node); break;
        case ASTNodeType::VarDecl:       visitVarDecl(node); break;
        case ASTNodeType::AssignStmt:    visitAssignStmt(node); break;
        case ASTNodeType::IfStmt:        visitIfStmt(node); break;
        case ASTNodeType::WhileStmt:     visitWhileStmt(node); break;
        case ASTNodeType::ReturnStmt:    visitReturnStmt(node); break;
        default:
            visitExpr(node);
            break;
    }
}
```

### 8.4 while循环代码生成

```cpp
void CodegenVisitor::visitWhileStmt(const shared_ptr<ASTNode>& node) {
    string L0 = sem_.newLabel();  // 循环开始标签
    string L1 = sem_.newLabel();  // 循环结束标签

    // 1. 输出循环开始标签
    sem_.emit("label", L0, "", "");

    // 2. 生成条件表达式代码
    string condAddr = visitExpr(node->children[0]);

    // 3. 条件为假时跳转到循环结束
    int j = sem_.emit("if_false", condAddr, "", L1);

    // 4. 生成循环体代码
    visit(node->children[1]);

    // 5. 无条件跳转到循环开始
    sem_.emit("goto", L0, "", "");

    // 6. 输出循环结束标签
    sem_.backpatch(j, L1);
    sem_.emit("label", L1, "", "");
}
```

### 8.5 表达式代码生成

```cpp
string CodegenVisitor::visitBinaryExpr(const shared_ptr<ASTNode>& node) {
    // 生成左操作数的代码
    string leftAddr = visitExpr(node->children[0]);
    // 生成右操作数的代码
    string rightAddr = visitExpr(node->children[1]);
    // 生成临时变量存储结果
    string t = sem_.newTemp();
    // 生成运算四元式
    sem_.emit(node->value, leftAddr, rightAddr, t);
    return t;
}
```

```cpp
string translateArithmetic(const string& command) {
    // add: SP--, D=M, A--, M=M+D
    if (command == "add")
        return "@SP\nAM=M-1\nD=M\nA=A-1\nM=M+D\n";

    // sub: SP--, D=M, A--, M=M-D
    if (command == "sub")
        return "@SP\nAM=M-1\nD=M\nA=A-1\nM=M-D\n";

    // neg: A=SP-1, M=-M
    if (command == "neg")
        return "@SP\nA=M-1\nM=-M\n";

    // eq/gt/lt: 使用跳转实现比较
    string t = generateLabel("T"), e = generateLabel("E");
    string jmp = (command == "eq") ? "D;JEQ" :
                 (command == "gt") ? "D;JGT" : "D;JLT";
    return "@SP\nAM=M-1\nD=M\nA=A-1\nD=M-D\n@" + t + "\n" + jmp +
           "\n@SP\nA=M-1\nM=0\n@" + e + "\n0;JMP\n(" + t +
           ")\n@SP\nA=M-1\nM=-1\n(" + e + ")\n";
}
```

### 12.5 函数调用翻译

```cpp
string translateCall(const string& functionName, int nArgs) {
    string rl = generateLabel("R");
    stringstream ss;

    // 1. 保存返回地址
    ss << "@" << rl << "\nD=A\n@SP\nAM=M+1\nA=A-1\nM=D\n";

    // 2. 保存调用者的栈帧
    ss << "@LCL\nD=M\n@SP\nAM=M+1\nA=A-1\nM=D\n";
    ss << "@ARG\nD=M\n@SP\nAM=M+1\nA=A-1\nM=D\n";
    ss << "@THIS\nD=M\n@SP\nAM=M+1\nA=A-1\nM=D\n";
    ss << "@THAT\nD=M\n@SP\nAM=M+1\nA=A-1\nM=D\n";

    // 3. 设置新的ARG
    ss << "@SP\nD=M\n@5\nD=D-A\n@" << nArgs << "\nD=D-A\n@ARG\nM=D\n";

    // 4. 设置新的LCL
    ss << "@SP\nD=M\n@LCL\nM=D\n";

    // 5. 跳转到函数
    ss << "@" << functionName << "\n0;JMP\n";

    // 6. 返回标签
    ss << "(" << rl << ")\n";

    return ss.str();
}
```

```cpp
string translateReturn() {
    // endFrame = LCL → R13
    // retAddr = *(endFrame - 5) → R14
    return
        "@LCL\nD=M\n@R13\nM=D\n"           // R13 = endFrame
        "@5\nA=D-A\nD=M\n@R14\nM=D\n"      // R14 = retAddr
        "@SP\nAM=M-1\nD=M\n@ARG\nA=M\nM=D\n"  // *ARG = pop()
        "@ARG\nD=M+1\n@SP\nM=D\n"           // SP = ARG+1
        "@R13\nAM=M-1\nD=M\n@THAT\nM=D\n"   // THAT = *(ef-1)
        "@R13\nAM=M-1\nD=M\n@THIS\nM=D\n"   // THIS = *(ef-2)
        "@R13\nAM=M-1\nD=M\n@ARG\nM=D\n"    // ARG  = *(ef-3)
        "@R13\nAM=M-1\nD=M\n@LCL\nM=D\n"    // LCL  = *(ef-4)
```
